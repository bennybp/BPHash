53  "Type to convert to must be an integral type");
54 
55  const size_t wantedbytes = sizeof(T);
56 
57  // will may pad with zero if needed
58  HashValue tmphash = truncate_hash(hash, wantedbytes);
59 
60  T ret = static_cast<T>(0);
