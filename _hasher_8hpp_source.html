262 /*! \brief Convenience function hashing selected elements of a container
263  *
264  * This can be used to easily obtain the hash of a range of objects
265  * in a container (given via two iterators).
266  *
267  * Objects in the range [first, last) are hashed (generally the same convention
268  * as the STL)
269  *
270  * \param [in] type The type of hash to use
271  * \param [in] first An iterator of the first object to hash
272  * \param [in] last An iterator of the element following the last element to hash
273  * \return Hash of the given data
274  */
275 template<typename InputIterator>
276 HashValue make_hash_range(HashType type, InputIterator first, InputIterator last)
277 {
278  Hasher hasher(type);
279 
280  for(auto it = first; it != last; ++it)
281  hasher(*it);
282 
283  return hasher.finalize();
