58 void MurmurHash3_128_x64::update(void const * data, size_t nbytes)
59 {
60  if(nbytes == 0)
61  return; // got nothing to do
62 
63  // cast to an array of bytes
64  const uint8_t * data_conv = static_cast<const uint8_t*>(data);
65 
66  if(nbuffer_ != 0)
67  {
68  // we have some leftover data. Add some from the
69  // new data and hash the temporary buffer
70 
71  // Amount of space left in the buffer?
72  size_t nbytes_avail = 16 - nbuffer_;
73 
74  // How much of data should we actually copy?
75  size_t tocopy = std::min(nbytes, nbytes_avail);
76 
77  // copy it to the end of the buffer
78  std::copy(data_conv,
79  data_conv + tocopy,
80  buffer_.begin() + nbuffer_);
81 
82  // how much do we have in the buffer now?
83  nbuffer_ += tocopy;
84 
85  // The new number of bytes to do in data
86  // This should not underflow since
87  // tocopy = std::min(nbytes, ...)
88  nbytes -= tocopy;
89 
90  // Also advance the data pointer
91  data_conv += tocopy;
92 
93  // hash the buffer if it is full
94  if(nbuffer_ == 16)
95  {
96  update_block_(buffer_.data(), 1);
97  nbuffer_ = 0;
98  }
99  }
100 
101  // now continue hashing the data in place
102  size_t nblocks = nbytes / 16;
103  update_block_(data_conv, nblocks); // ok if nblocks == 0
104 
105  // advance the pointer and calculate how much is left
106  nbytes -= (nblocks * 16);
107  data_conv += (nblocks * 16);
108 
109  // Leave any remainder in the main buffer
110  // (we already know that nbytes < 16, or else
111  // the while loop would have kept going)
112  if(nbytes != 0)
113  {
114  std::copy(data_conv, data_conv + nbytes, buffer_.begin());
115  nbuffer_ = nbytes;
116  }
117 }

193 void MurmurHash3_128_x64::update_block_(uint8_t const * data, size_t nblocks)
194 {
